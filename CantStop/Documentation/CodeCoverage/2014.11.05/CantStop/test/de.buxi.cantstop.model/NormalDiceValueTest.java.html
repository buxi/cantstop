	@Test

		assertEquals("Value 6 should be", 6, throwDiceOntoX(6));
	}

	/**
	 * @param targetValue 
	 * @return
	 */
	private int throwDiceOntoX(int targetValue) {
		dice.reset();
		int number = 0;
		int i = maxthrows; 
		do {
			dice.throwDice();
			number = dice.getDiceValue(); 
			i--;
			
		} while( number != targetValue && i > 0);
		return number;
	}
